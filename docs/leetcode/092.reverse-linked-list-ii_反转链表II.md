#### [92. 反转链表 II ](https://leetcode.cn/problems/reverse-linked-list-ii/)

难度中等

给你单链表的头指针 `head` 和两个整数 `left` 和 `right` ，其中 `left <= right` 。请你反转从位置 `left` 到位置 `right` 的链表节点，返回 **反转后的链表** 。

**示例 1：**

![](https://assets.leetcode.com/uploads/2021/02/19/rev2ex2.jpg)

**输入：** head = [1,2,3,4,5], left = 2, right = 4

**输出：**[1,4,3,2,5]

**示例 2：**

**输入：** head = [5], left = 1, right = 1

**输出：**[5]

**提示：**

-   链表中节点数目为 `n`
-   `1 <= n <= 500`
-   `-500 <= Node.val <= 500`
-   `1 <= left <= right <= n`

**进阶：** 你可以使用一趟扫描完成反转吗？

    // 先前进left次，找到prev和curr；进行right-left次头插得到
    ListNode* reverseBetween(ListNode* head, int left, int right) {
        ListNode * dummy = new ListNode(-1,head);
        ListNode * prev = dummy;
        for(int i=1; i< left;++i){
            prev = prev->next;
        }
        ListNode * curr = prev->next;
        // p c n
        // p n c
        for(int i =left; i< right;++i){
            ListNode * next = curr->next;
            curr->next = curr->next->next;
            next->next = prev->next;
            prev->next = next;
        }
        return dummy->next;
    }

    ListNode *next = nullptr;
    ListNode * reverse(ListNode *curr, int i, int n) {
        if (curr==nullptr || curr->next==nullptr || i == n) {
            if (curr) {
                next = curr->next;
            }
            return curr;
        }
        ListNode *newhead =reverse(curr->next, i+1, n);
        curr->next->next = curr;
        curr->next = nullptr;
        return newhead;
    }

// 原始记录使用dummy记录首节点，
// 同时，使用start记录原始 开始start和结束end节点，用于最后反转后的连接
    (start) (end) ?
    --- A -> a -> b-> c -> Z --- ---

    ---> ? <- a <- b<- c ? --- ---
    ---> A -> c -> b-> a -> Z --- ---
    // 这里应该将a的前一个节点A作为pre,然后这个pre A的next应该是结束节点end(c)；
    // 然后a的下一个节点应该是本来的end(c)的下一个节点Z;
    // 所以最后反转后,start 也就是 pre A的 next 为 c (最后一个节点);
    // 反转后的节点a的下一个节点next为 Z;


ListNode* reverseBetween(ListNode* head, int left, int right) {
    ListNode* dummy = new ListNode(-1);
    dummy->next = head;
    ListNode* pre = dummy;
    ListNode* cur = head;
    ListNode* next = nullptr;
    for (int i = 1; i < left; i++) {
        pre = pre->next;
    }
    cur = pre->next;
    ListNode* end = cur;
    ListNode* start = pre;
    for (int i = left; i <= right; i++) {
        next = cur->next;
        cur->next = pre;
        pre = cur;
        cur = next;
    }
    start->next = pre;
    end->next = cur;
    return dummy->next;
}

 最后         pre  cur

 tail -> next -> next = cur
 tail -> next = pre

class Solution {
public:
    ListNode* reverseBetween(ListNode* head, int left, int right) {
        ListNode * dummy = new ListNode(-1,head);
        ListNode * tail = dummy;
        ListNode * curr = dummy;
        int i = 1;
        while (i < left){
            tail = tail->next;
            i++;
        }
        ListNode * prev = nullptr;
        ListNode * next = nullptr;
        curr = tail->next;
        while(i <= right) {
            next = curr->next;
            curr->next = prev;
            prev = curr;
            curr = next;
            ++i;
        }
        tail->next->next = curr;
        tail->next = prev;
        return dummy->next;
    }
};

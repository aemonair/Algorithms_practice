```cpp
class Solution {
public:
    int deleteAndEarn(vector<int>& nums) {
        int n = ranges::max(nums);
        vector<int> all(n+1);
        for (auto x: nums) {
            all[x]++;
        }
        vector dp(n+1, 0);
        dp[1] = all[1];
        for (int i = 2; i <= n; ++i) {
            dp[i] = max(dp[i-1], dp[i-2]+all[i]*i);
        }
        return dp[n];
    }
};
```

```cpp
class Solution {
public:
    int deleteAndEarn(vector<int>& nums) {
        int n = ranges::max(nums);

        vector<int> cnt(n+1);
        for (auto num: nums) {
            cnt[num]++;
        }

        vector<int> dp(n+3);
        for (int i = 0; i < n+1; ++i) {
            dp[i+2] = max(dp[i] + cnt[i]*i , dp[i+1]);
        }
        return dp[n+2];
    }
};
```

```cpp
class Solution {
public:
    int deleteAndEarn(vector<int>& nums) {
        int n = ranges::max(nums);
        vector<int> vec(n+1);
        for (auto num: nums) {
            vec[num]++;
        }

        vector<int> dp(n+3);
        for (int i = 2; i <= n+2; i++) {
            dp[i] = max(dp[i-1], dp[i-2] + (i-2) * vec[i-2]);
        }
        return ranges::max(dp);
    }
};
```
